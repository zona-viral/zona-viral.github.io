const ADS = { main: "https://omg10.com/4/10902178", back: "https://omg10.com/4/10830632" }; const video = document.getElementById("video"); let mainShown = false; let backShown = false; /* 🎬 TRIGGER SAAT VIDEO DIKLIK / DIPLAY */ function triggerMainAd() { if (!mainShown) { window.open(ADS.main, "_blank"); mainShown = true; } } /* CLICK VIDEO */ video.addEventListener("click", triggerMainAd); /* PLAY EVENT (lebih penting) */ video.addEventListener("play", triggerMainAd); /* BACK AD */ window.addEventListener("focus", () => { if (mainShown && !backShown) { backShown = true; setTimeout(() => { window.open(ADS.back, "_blank"); }, 1500); } });
